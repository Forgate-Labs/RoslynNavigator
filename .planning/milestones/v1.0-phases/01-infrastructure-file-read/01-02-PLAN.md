Create `src/RoslynNavigator/Commands/FileReadCommand.cs` with namespace `RoslynNavigator.Commands`:

```
using RoslynNavigator.Models;

public static class FileReadCommand
{
    public static async Task<FileReadResult> ExecuteAsync(string path, string? lines)
```

Implementation:
1. Resolve `path` to absolute: `Path.GetFullPath(path)` (or combine with `Directory.GetCurrentDirectory()` if relative)
2. If file does not exist, throw `FileNotFoundException($"File not found: {path}")`
3. Read all lines: `var allLines = await File.ReadAllLinesAsync(absolutePath)`
4. Set `totalLines = allLines.Length`
5. Parse `lines` parameter (format: "START-END", e.g., "10-20"):
   - If null/empty: rangeStart = 1, rangeEnd = totalLines (return all)
   - If provided: split on '-', parse int start and int end (1-based, inclusive)
   - Clamp: start = Math.Max(1, start), end = Math.Min(totalLines, end)
   - If start > end: throw `ArgumentException("Invalid line range: start must be <= end")`
6. Slice lines array: `allLines[(rangeStart-1)..(rangeEnd)]` (convert to 0-based index)
7. Build `FileLineInfo` list: for each sliced line at index `i`, `Line = rangeStart + i`, `Content = line`
8. Return `FileReadResult { FilePath = path, TotalLines = totalLines, RangeStart = (lines != null ? rangeStart : null), RangeEnd = (lines != null ? rangeEnd : null), Lines = lineInfos }`

```
using RoslynNavigator.Models;
using System.Text.RegularExpressions;

public static class FileGrepCommand
{
    public static async Task<FileGrepResult> ExecuteAsync(string pattern, string? searchPath, string? ext, int maxLines)
```

Implementation:
1. Compile regex: `var regex = new Regex(pattern, RegexOptions.Compiled | RegexOptions.IgnoreCase)`
   - Wrap in try/catch — if regex is invalid, throw `ArgumentException($"Invalid regex pattern: {ex.Message}")`
2. Resolve `searchPath`: if null, use `Directory.GetCurrentDirectory()`. Resolve to absolute.
3. Determine files to search:
   - If `searchPath` points to a file: `files = new[] { absolutePath }`
   - If `searchPath` points to a directory: `files = Directory.GetFiles(absolutePath, "*", SearchOption.AllDirectories)`
     - If `ext` is not null (e.g., ".cs"): filter `files` where `Path.GetExtension(f).Equals(ext, StringComparison.OrdinalIgnoreCase)`
4. Search files: for each file, `var fileLines = await File.ReadAllLinesAsync(file)`. For each line at index `i` where `regex.IsMatch(line)`:
   - Add `GrepMatch { FilePath = file (relative to CWD via Path.GetRelativePath(Directory.GetCurrentDirectory(), file)), Line = i + 1, Content = line.Trim() }`
   - If `matches.Count >= maxLines`: set `truncated = true` and break out of outer loop
5. Return `FileGrepResult { Pattern = pattern, TotalMatches = matches.Count, Truncated = truncated, Matches = matches }`

```csharp
// ── file command group ──────────────────────────────────────────────────────
var fileCommand = new Command("file", "File read and staged-edit operations");

// file read
var fileReadPathArg = new Argument<string>("path", "Path to the file to read");
var fileReadLinesOption = new Option<string?>("--lines", "Line range to read (e.g., 10-20)");
var fileReadSubcommand = new Command("read", "Read a file with line numbers");
fileReadSubcommand.AddArgument(fileReadPathArg);
fileReadSubcommand.AddOption(fileReadLinesOption);
fileReadSubcommand.SetHandler(async (string path, string? lines) =>
{
    try
    {
        var result = await FileReadCommand.ExecuteAsync(path, lines);
        Console.WriteLine(JsonSerializer.Serialize(result, jsonOptions));
    }
    catch (Exception ex)
    {
        OutputError("file_read_error", ex.Message);
        Environment.ExitCode = 1;
    }
}, fileReadPathArg, fileReadLinesOption);

// file grep
var fileGrepPatternArg = new Argument<string>("pattern", "Regex pattern to search for");
var fileGrepPathArg = new Argument<string?>("path", "File or directory to search (default: current directory)") { Arity = ArgumentArity.ZeroOrOne };
var fileGrepExtOption = new Option<string?>("--ext", "File extension filter (e.g., .cs)");
var fileGrepMaxLinesOption = new Option<int>("--max-lines", () => 100, "Maximum number of matching lines to return");
var fileGrepSubcommand = new Command("grep", "Search for a regex pattern in files");
fileGrepSubcommand.AddArgument(fileGrepPatternArg);
fileGrepSubcommand.AddArgument(fileGrepPathArg);
fileGrepSubcommand.AddOption(fileGrepExtOption);
fileGrepSubcommand.AddOption(fileGrepMaxLinesOption);
fileGrepSubcommand.SetHandler(async (string pattern, string? path, string? ext, int maxLines) =>
{
    try
    {
        var result = await FileGrepCommand.ExecuteAsync(pattern, path, ext, maxLines);
        Console.WriteLine(JsonSerializer.Serialize(result, jsonOptions));
    }
    catch (Exception ex)
    {
        OutputError("file_grep_error", ex.Message);
        Environment.ExitCode = 1;
    }
}, fileGrepPatternArg, fileGrepPathArg, fileGrepExtOption, fileGrepMaxLinesOption);

fileCommand.AddCommand(fileReadSubcommand);
fileCommand.AddCommand(fileGrepSubcommand);
```
