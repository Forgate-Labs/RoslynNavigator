  <behavior>
    Method signatures:
      `UpdateMemberResult UpdateMember(string sourceText, string typeName, string memberKind, string memberName, string newContent)`
      `RemoveMemberResult RemoveMember(string sourceText, string typeName, string memberKind, string memberName)`

    UpdateMemberResult record:
      `bool Success`
      `string ModifiedSource`   // full modified file content (unchanged if failed)
      `string? Error`           // null on success

    RemoveMemberResult record:
      `bool Success`
      `string ModifiedSource`   // full modified file content (unchanged if failed)
      `string? Error`           // null on success

    memberKind values for UpdateMember: "property" | "field"
    memberKind values for RemoveMember: "method" | "property" | "field"
    target types: ClassDeclarationSyntax, RecordDeclarationSyntax, StructDeclarationSyntax

    UpdateMember — lookup rules:
      - "property": scan typeDecl.Members for PropertyDeclarationSyntax where Identifier.Text == memberName
      - "field": scan typeDecl.Members for FieldDeclarationSyntax where any declarator.Identifier.Text == memberName
        (field names strip leading underscore for comparison: "_count" matches memberName "count" OR "_count")
      - Validate newContent by wrapping in temp class, checking CSharpSyntaxTree.ParseText diagnostics — error if invalid syntax
      - Replace the found node with parsed newContent (re-use indentation detection from DotnetAddMemberService pattern)
      - If not found: return UpdateMemberResult { Success=false, Error="Member '{memberName}' of kind '{memberKind}' not found in type '{typeName}'" }

    RemoveMember — lookup rules:
      - "method": scan for MethodDeclarationSyntax where Identifier.Text == memberName
      - "property": scan for PropertyDeclarationSyntax where Identifier.Text == memberName
      - "field": scan for FieldDeclarationSyntax where any declarator.Identifier.Text == memberName
        (same underscore-tolerant matching as UpdateMember)
      - Remove the node: typeDecl.WithMembers(typeDecl.Members.Remove(foundMember))
      - If not found: return RemoveMemberResult { Success=false, Error="Member '{memberName}' of kind '{memberKind}' not found in type '{typeName}'" }

    Test cases (write these as xUnit [Fact] tests — RED first, then GREEN):
      UpdateMember:
        - UpdateMember(classWithProperty, "MyClass", "property", "Name", "public string Name { get; init; }") → property replaced, old getter/setter gone
        - UpdateMember(classWithField, "MyClass", "field", "_count", "private long _count;") → field replaced
        - UpdateMember(classWithField, "MyClass", "field", "count", "private long _count;") → same result (underscore-tolerant match)
        - UpdateMember(classWithProperty, "MyClass", "property", "Missing", "...") → Error contains "not found"
        - UpdateMember(classWithProperty, "MyClass", "property", "Name", "NOT VALID C# {{{{") → Error contains "syntax error"
        - UpdateMember(structSource, "MyStruct", "property", "Value", "public int Value { get; init; }") → works for struct
        - UpdateMember(recordSource, "MyRecord", "property", "Id", "public int Id { get; init; }") → works for record

      RemoveMember:
        - RemoveMember(classWithAll, "MyClass", "method", "ExistingMethod") → method removed, rest intact
        - RemoveMember(classWithAll, "MyClass", "property", "Name") → property removed
        - RemoveMember(classWithAll, "MyClass", "field", "_existingField") → field removed
        - RemoveMember(classWithAll, "MyClass", "field", "existingField") → same (underscore-tolerant)
        - RemoveMember(classWithAll, "MyClass", "method", "Nonexistent") → Error contains "not found"
        - RemoveMember(structSource, "MyStruct", "field", "_x") → works for struct
  </behavior>

      private static bool FieldNameMatches(FieldDeclarationSyntax field, string memberName)

          var normalized = memberName.TrimStart('_');
          return field.Declaration.Variables.Any(v =>
              v.Identifier.Text == memberName ||                   // exact match ("_count")
              v.Identifier.Text.TrimStart('_') == normalized);    // normalized match ("count")
      }
